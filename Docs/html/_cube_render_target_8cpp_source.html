81void CubeRenderTarget::BuildDescriptors()
82{
83 //step5: We now need to create an SRV to the cube map resource so that we can sample it in a pixel shader after it is built
84 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
85 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
86 srvDesc.Format = mFormat;
87 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURECUBE;
88 srvDesc.TextureCube.MostDetailedMip = 0;
89 srvDesc.TextureCube.MipLevels = 1;
90 srvDesc.TextureCube.ResourceMinLODClamp = 0.0f;
91
92 // Create SRV to the entire cubemap resource.
93 md3dDevice->CreateShaderResourceView(mCubeMap.Get(), &srvDesc, mhCpuSrv);
94
95 // step 6: We also need to create a render target view to each element in the cube map texture array, so that we can render onto each cube map face one by one.
96 for(int i = 0; i < 6; ++i)
97 {
98 D3D12_RENDER_TARGET_VIEW_DESC rtvDesc;
99 rtvDesc.ViewDimension = D3D12_RTV_DIMENSION_TEXTURE2DARRAY;
100 rtvDesc.Format = mFormat;
101 rtvDesc.Texture2DArray.MipSlice = 0;
102 rtvDesc.Texture2DArray.PlaneSlice = 0;
103
104 // Render target to ith element.
105 rtvDesc.Texture2DArray.FirstArraySlice = i;
106
107 // Only view one element of the array.
108 rtvDesc.Texture2DArray.ArraySize = 1;
109
110 // Create RTV to ith cubemap face.
111 md3dDevice->CreateRenderTargetView(mCubeMap.Get(), &rtvDesc, mhCpuRtv[i]);
112 }
113}

116void CubeRenderTarget::BuildResource()
117{
118 // Note, compressed formats cannot be used for UAV. We get error like:
119 // ERROR: ID3D11Device::CreateTexture2D: The format (0x4d, BC3_UNORM)
120 // cannot be bound as an UnorderedAccessView, or cast to a format that
121 // could be bound as an UnorderedAccessView. Therefore this format
122 // does not support D3D11_BIND_UNORDERED_ACCESS.
123
124 D3D12_RESOURCE_DESC texDesc;
125 ZeroMemory(&texDesc, sizeof(D3D12_RESOURCE_DESC));
126 texDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
127 texDesc.Alignment = 0;
128 texDesc.Width = mWidth;
129 texDesc.Height = mHeight;
130 //step2: Creating a cube map texture is done by creating a texture array with six elements (one for each face).
131 texDesc.DepthOrArraySize = 6;
132 texDesc.MipLevels = 1;
133 texDesc.Format = mFormat;
134 texDesc.SampleDesc.Count = 1;
135 texDesc.SampleDesc.Quality = 0;
136 texDesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
137 //step3: Because we are going to render to the cube map, we must set the D3D12_RESOURCE_FLAG_ALLOW_RENDER_TARGET flag.
138 texDesc.Flags = D3D12_RESOURCE_FLAG_ALLOW_RENDER_TARGET;
139
140 ThrowIfFailed(md3dDevice->CreateCommittedResource(
141 &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_DEFAULT),
142 D3D12_HEAP_FLAG_NONE,
143 &texDesc,
144 D3D12_RESOURCE_STATE_GENERIC_READ,
145 nullptr,
146 IID_PPV_ARGS(&mCubeMap)));
147}
