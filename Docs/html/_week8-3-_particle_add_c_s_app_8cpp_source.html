153void ParticleAddCSApp::DoComputeWork()
154{
155 // Reuse the memory associated with command recording.
156 // We can only reset when the associated command lists have finished execution on the GPU.
157 ThrowIfFailed(mDirectCmdListAlloc->Reset());
158
159 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
160 // Reusing the command list reuses memory.
161 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), mPSOs["particleAdd"].Get()));
162
163 ID3D12DescriptorHeap* descriptorHeaps[] = { mUavHeap.Get() };
164 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
165
166
167 mCommandList->SetComputeRootSignature(mRootSignature.Get());
168
169 //mCommandList->SetComputeRootUnorderedAccessView(0, mInputBufferA->GetGPUVirtualAddress());
170 //mCommandList->SetComputeRootUnorderedAccessView(1, mOutputBuffer->GetGPUVirtualAddress());
171
172 int passUavIndex = 0;
173 auto passUavHandle = CD3DX12_GPU_DESCRIPTOR_HANDLE(mUavHeap->GetGPUDescriptorHandleForHeapStart());
174
175 mCommandList->SetComputeRootDescriptorTable(0, passUavHandle); //mInputBufferA->GetGPUVirtualAddress()
176 passUavHandle.Offset(passUavIndex, mCbvSrvUavDescriptorSize);
177 mCommandList->SetComputeRootDescriptorTable(1, passUavHandle); //mOutputBuffer->GetGPUVirtualAddress()
178
179 mCommandList->Dispatch(1, 1, 1);
180
181 // Schedule to copy the data to the default buffer to the readback buffer.
182
183 //D3D12_RESOURCE_STATE_COPY_SOURCE: The resource is used as the source in a copy operation.
184 //D3D12_RESOURCE_STATE_COPY_DEST: The resource is used as the destination in a copy operation.
185
186 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(mOutputBuffer.Get(),
187 D3D12_RESOURCE_STATE_UNORDERED_ACCESS, D3D12_RESOURCE_STATE_COPY_SOURCE));
188
189 //We need to read the data back from the GPU after computer shader does its calculation
190 //mCommandList->CopyResource(Destination: CPUData, Source: GPUData);
191 mCommandList->CopyResource(mReadBackBuffer.Get(), mOutputBuffer.Get());
192
193 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(mOutputBuffer.Get(),
194 D3D12_RESOURCE_STATE_COPY_SOURCE, D3D12_RESOURCE_STATE_UNORDERED_ACCESS));
195
196
197 // Done recording commands.
198 ThrowIfFailed(mCommandList->Close());
199
200 // Add the command list to the queue for execution.
201 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
202 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
203
204 // Wait for the work to finish.
205 //FlushCommandQueue();
206
207 // Map the data so we can read it on CPU.
208 Particle* mappedData = nullptr;
209 CD3DX12_RANGE readRange(0, 0); // We do not intend to read from this resource on the CPU.
210 ThrowIfFailed(mReadBackBuffer->Map(0, nullptr, reinterpret_cast<void**>(&mappedData)));
211
212 std::ofstream fout("results.txt");
213
214 for (int i = 0; i < NumDataElements; ++i)
215 {
216 fout << "(" << mappedData[i].Acceleration.x << ", " << mappedData[i].Acceleration.y << ", " << mappedData[i].Acceleration.z <<
217 ", " << mappedData[i].Position.x << ", " << mappedData[i].Position.y << ", " << mappedData[i].Position.z <<
218 ", " << mappedData[i].Velocity.x << ", " << mappedData[i].Velocity.y << ", " << mappedData[i].Velocity.z << ")" << std::endl;
219 }
220
221 mReadBackBuffer->Unmap(0, nullptr);
222}

224void ParticleAddCSApp::BuildBuffers()
225{
226 // Generate some data.
227 std::vector<Particle> dataA(NumDataElements);
228 for (int i = 0; i < NumDataElements; ++i)
229 {
230 dataA[i].Acceleration = XMFLOAT3(1.0f, 1.0f, 1.0f);
231 dataA[i].Position = XMFLOAT3((float)i, (float)i, (float)i);
232 dataA[i].Velocity = XMFLOAT3(10.0f, 10.0f, 0);
233 }
234
235 UINT64 byteSize = dataA.size() * sizeof(Particle);
236
237
238 // Create the actual input buffer resource.
239 ThrowIfFailed(md3dDevice->CreateCommittedResource(
240 &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_DEFAULT),
241 D3D12_HEAP_FLAG_NONE,
242 &CD3DX12_RESOURCE_DESC::Buffer(byteSize, D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS),
243 D3D12_RESOURCE_STATE_UNORDERED_ACCESS,
244 nullptr,
245 IID_PPV_ARGS(&mInputBufferA)));
246
247 // In order to copy CPU memory data into our default buffer, we need to create
248 // an intermediate upload heap.
249 ThrowIfFailed(md3dDevice->CreateCommittedResource(
250 &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_UPLOAD),
251 D3D12_HEAP_FLAG_NONE,
252 &CD3DX12_RESOURCE_DESC::Buffer(byteSize),
253 D3D12_RESOURCE_STATE_GENERIC_READ,
254 nullptr,
255 IID_PPV_ARGS(mInputUploadBufferA.GetAddressOf())));
256
257
258 // Describe the data we want to copy into the default buffer.
259 D3D12_SUBRESOURCE_DATA subResourceData = {};
260 subResourceData.pData = dataA.data();
261 subResourceData.RowPitch = byteSize;
262 subResourceData.SlicePitch = subResourceData.RowPitch;
263
264 // Schedule to copy the data to the default buffer resource. At a high level, the helper function UpdateSubresources
265 // will copy the CPU memory into the intermediate upload heap. Then, using ID3D12CommandList::CopySubresourceRegion,
266 // the intermediate upload heap data will be copied to mBuffer.
267 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(mInputBufferA.Get(),
268 D3D12_RESOURCE_STATE_UNORDERED_ACCESS, D3D12_RESOURCE_STATE_COPY_DEST));
269
270 UpdateSubresources(mCommandList.Get(), mInputBufferA.Get(), mInputUploadBufferA.Get(), 0, 0, 1, &subResourceData);
271
272
273 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(mInputBufferA.Get(),
274 D3D12_RESOURCE_STATE_COPY_DEST, D3D12_RESOURCE_STATE_UNORDERED_ACCESS));
275
276
277 // Create the buffer that will be a UAV.
278 ThrowIfFailed(md3dDevice->CreateCommittedResource(
279 &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_DEFAULT),
280 D3D12_HEAP_FLAG_NONE,
281 &CD3DX12_RESOURCE_DESC::Buffer(byteSize, D3D12_RESOURCE_FLAG_ALLOW_UNORDERED_ACCESS),
282 D3D12_RESOURCE_STATE_UNORDERED_ACCESS,
283 nullptr,
284 IID_PPV_ARGS(&mOutputBuffer)));
285
286 //Specifies a heap used for reading back. This heap type has CPU access optimized for reading data back from the GPU,
287 //but does not experience the maximum amount of bandwidth for the GPU. This heap type is best for GPU-write-once,
288 //CPU-readable data. The CPU cache behavior is write-back, which is conducive for multiple sub-cache-line CPU reads.
289 //Resources in this heap must be created with D3D12_RESOURCE_STATE_COPY_DEST, and cannot be changed away from this
290 ThrowIfFailed(md3dDevice->CreateCommittedResource(
291 &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_READBACK),
292 D3D12_HEAP_FLAG_NONE,
293 &CD3DX12_RESOURCE_DESC::Buffer(byteSize),
294 D3D12_RESOURCE_STATE_COPY_DEST,
295 nullptr,
296 IID_PPV_ARGS(&mReadBackBuffer)));
297}

299void ParticleAddCSApp::BuildRootSignature()
300{
301
302
303 CD3DX12_DESCRIPTOR_RANGE uavTable0;
304 uavTable0.Init(D3D12_DESCRIPTOR_RANGE_TYPE_UAV, 1, 0);
305
306 CD3DX12_DESCRIPTOR_RANGE uavTable1;
307 uavTable1.Init(D3D12_DESCRIPTOR_RANGE_TYPE_UAV, 1, 1);
308
309 // Root parameter can be a table, root descriptor or root constants.
310 CD3DX12_ROOT_PARAMETER slotRootParameter[2];
311
312 // Create root CBVs.
313 slotRootParameter[0].InitAsDescriptorTable(1, &uavTable0);
314 slotRootParameter[1].InitAsDescriptorTable(1, &uavTable1);
315
316
317 // A root signature is an array of root parameters.
318 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(2, slotRootParameter,
319 0, nullptr,
320 D3D12_ROOT_SIGNATURE_FLAG_NONE);
321
322 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
323 ComPtr<ID3DBlob> serializedRootSig = nullptr;
324 ComPtr<ID3DBlob> errorBlob = nullptr;
325
326
327 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
328 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
329
330 if (errorBlob != nullptr)
331 {
332 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
333 }
334 ThrowIfFailed(hr);
335
336 ThrowIfFailed(md3dDevice->CreateRootSignature(
337 0,
338 serializedRootSig->GetBufferPointer(),
339 serializedRootSig->GetBufferSize(),
340 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
341}
