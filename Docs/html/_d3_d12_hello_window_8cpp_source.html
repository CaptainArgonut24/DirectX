29// Load the rendering pipeline dependencies.
30void D3D12HelloWindow::LoadPipeline()
31{
32 UINT dxgiFactoryFlags = 0;
33
34#if defined(_DEBUG)
35 // Step2: Enable the debug layer (requires the Graphics Tools "optional feature").
36 // NOTE: Enabling the debug layer after device creation will invalidate the active device.
37 {
38 ComPtr<ID3D12Debug> debugController;
39 if (SUCCEEDED(D3D12GetDebugInterface(IID_PPV_ARGS(&debugController))))
40 {
41 debugController->EnableDebugLayer();
42
43 // Enable additional debug layers.
44 dxgiFactoryFlags |= DXGI_CREATE_FACTORY_DEBUG;
45 }
46 }
47#endif
48
49 //step3: Enables creating Microsoft DirectX Graphics Infrastructure (DXGI) objects
50 ComPtr<IDXGIFactory4> factory;
51 ThrowIfFailed(CreateDXGIFactory2(dxgiFactoryFlags, IID_PPV_ARGS(&factory)));
52
53 //step4: create a Windows Advanced Rasterization Platform (WARP) WARP enables Rendering When Direct3D Hardware is Not Available
54
55 if (m_useWarpDevice)
56 {
57 ComPtr<IDXGIAdapter> warpAdapter;
58 ThrowIfFailed(factory->EnumWarpAdapter(IID_PPV_ARGS(&warpAdapter)));
59
60 ThrowIfFailed(D3D12CreateDevice(
61 warpAdapter.Get(),
62 D3D_FEATURE_LEVEL_11_0,
63 IID_PPV_ARGS(&m_device)
64 ));
65 }
66 else
67 {
68 //step5: The IDXGIAdapter interface represents a display subsystem (including one or more GPUs, DACs and video memory).
69 ComPtr<IDXGIAdapter1> hardwareAdapter;
70 GetHardwareAdapter(factory.Get(), &hardwareAdapter);
71
72 //step6: Creates a device that represents the display adapter.
73 ThrowIfFailed(D3D12CreateDevice(
74 hardwareAdapter.Get(),
75 D3D_FEATURE_LEVEL_11_0,
76 IID_PPV_ARGS(&m_device)
77 ));
78 }
79
80 // step7: Describe and create the command queue. Fill out a command queue description, then create the command queue:
81 D3D12_COMMAND_QUEUE_DESC queueDesc = {};
82 queueDesc.Flags = D3D12_COMMAND_QUEUE_FLAG_NONE;
83 queueDesc.Type = D3D12_COMMAND_LIST_TYPE_DIRECT;
84
85 ThrowIfFailed(m_device->CreateCommandQueue(&queueDesc, IID_PPV_ARGS(&m_commandQueue)));
86
87 // step8: Describe and create the swap chain. Fill out a swapchain description, then create the swap chain
88 DXGI_SWAP_CHAIN_DESC1 swapChainDesc = {};
89 swapChainDesc.BufferCount = FrameCount;
90 swapChainDesc.Width = m_width;
91 swapChainDesc.Height = m_height;
92 swapChainDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
93 swapChainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
94 swapChainDesc.SwapEffect = DXGI_SWAP_EFFECT_FLIP_DISCARD;
95 swapChainDesc.SampleDesc.Count = 1;
96
97 ComPtr<IDXGISwapChain1> swapChain;
98 ThrowIfFailed(factory->CreateSwapChainForHwnd(
99 m_commandQueue.Get(), // Swap chain needs the queue so that it can force a flush on it.
100 Win32Application::GetHwnd(),
101 &swapChainDesc,
102 nullptr,
103 nullptr,
104 &swapChain
105 ));
106
107 // This sample does not support fullscreen transitions.
108 ThrowIfFailed(factory->MakeWindowAssociation(Win32Application::GetHwnd(), DXGI_MWA_NO_ALT_ENTER));
109
110 ThrowIfFailed(swapChain.As(&m_swapChain));
111 m_frameIndex = m_swapChain->GetCurrentBackBufferIndex();
112
113 // step9: Create descriptor heaps. Fill out a heap description. then create a descriptor heap:
114 {
115 // Describe and create a render target view (RTV) descriptor heap.
116 D3D12_DESCRIPTOR_HEAP_DESC rtvHeapDesc = {};
117 rtvHeapDesc.NumDescriptors = FrameCount;
118 rtvHeapDesc.Type = D3D12_DESCRIPTOR_HEAP_TYPE_RTV;
119 rtvHeapDesc.Flags = D3D12_DESCRIPTOR_HEAP_FLAG_NONE;
120 ThrowIfFailed(m_device->CreateDescriptorHeap(&rtvHeapDesc, IID_PPV_ARGS(&m_rtvHeap)));
121
122 m_rtvDescriptorSize = m_device->GetDescriptorHandleIncrementSize(D3D12_DESCRIPTOR_HEAP_TYPE_RTV);
123 }
124
125 // step10 : Create frame resources. Create the render target view:
126 {
127 CD3DX12_CPU_DESCRIPTOR_HANDLE rtvHandle(m_rtvHeap->GetCPUDescriptorHandleForHeapStart());
128
129 // Create a RTV for each frame.
130 for (UINT n = 0; n < FrameCount; n++)
131 {
132 ThrowIfFailed(m_swapChain->GetBuffer(n, IID_PPV_ARGS(&m_renderTargets[n])));
133 m_device->CreateRenderTargetView(m_renderTargets[n].Get(), nullptr, rtvHandle);
134 rtvHandle.Offset(1, m_rtvDescriptorSize);
135 }
136 }
137
138 ThrowIfFailed(m_device->CreateCommandAllocator(D3D12_COMMAND_LIST_TYPE_DIRECT, IID_PPV_ARGS(&m_commandAllocator)));
139}

141// Load the sample assets.
142void D3D12HelloWindow::LoadAssets()
143{
144 // Create the command list.
145 ThrowIfFailed(m_device->CreateCommandList(0, D3D12_COMMAND_LIST_TYPE_DIRECT, m_commandAllocator.Get(), nullptr, IID_PPV_ARGS(&m_commandList)));
146
147 // Command lists are created in the recording state, but there is nothing
148 // to record yet. The main loop expects it to be closed, so close it now.
149 ThrowIfFailed(m_commandList->Close());
150
151 // Create synchronization objects.
152 {
153 ThrowIfFailed(m_device->CreateFence(0, D3D12_FENCE_FLAG_NONE, IID_PPV_ARGS(&m_fence)));
154 m_fenceValue = 1;
155
156 // Create an event handle to use for frame synchronization.
157 m_fenceEvent = CreateEvent(nullptr, FALSE, FALSE, nullptr);
158 if (m_fenceEvent == nullptr)
159 {
160 ThrowIfFailed(HRESULT_FROM_WIN32(GetLastError()));
161 }
162 }
163}

170// Render the scene.
171void D3D12HelloWindow::OnRender()
172{
173 // Record all the commands we need to render the scene into the command list.
174 PopulateCommandList();
175
176 // Execute the command list.
177 ID3D12CommandList* ppCommandLists[] = { m_commandList.Get() };
178 m_commandQueue->ExecuteCommandLists(_countof(ppCommandLists), ppCommandLists);
179
180 // Present the frame.
181 ThrowIfFailed(m_swapChain->Present(1, 0));
182
183 WaitForPreviousFrame();
184}

188 // Ensure that the GPU is no longer referencing resources that are about to be

191

195void D3D12HelloWindow::PopulateCommandList()
196{
197 // Command list allocators can only be reset when the associated
198 // command lists have finished execution on the GPU; apps should use
199 // fences to determine GPU execution progress.
200 ThrowIfFailed(m_commandAllocator->Reset());
201
202 // However, when ExecuteCommandList() is called on a particular command
203 // list, that command list can then be reset at any time and must be before
204 // re-recording.
205 ThrowIfFailed(m_commandList->Reset(m_commandAllocator.Get(), m_pipelineState.Get()));
206
207 // Indicate that the back buffer will be used as a render target.
208 m_commandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(m_renderTargets[m_frameIndex].Get(), D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
209
210 CD3DX12_CPU_DESCRIPTOR_HANDLE rtvHandle(m_rtvHeap->GetCPUDescriptorHandleForHeapStart(), m_frameIndex, m_rtvDescriptorSize);
211
212 // Record commands.
213 const float clearColor[] = { 0.0f, 0.2f, 0.4f, 1.0f };
214 m_commandList->ClearRenderTargetView(rtvHandle, clearColor, 0, nullptr);
215
216 // Indicate that the back buffer will now be used to present.
217 m_commandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(m_renderTargets[m_frameIndex].Get(), D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
218
219 ThrowIfFailed(m_commandList->Close());
220}

222void D3D12HelloWindow::WaitForPreviousFrame()
223{
224 // WAITING FOR THE FRAME TO COMPLETE BEFORE CONTINUING IS NOT BEST PRACTICE.
225 // This is code implemented as such for simplicity. The D3D12HelloFrameBuffering
226 // sample illustrates how to use fences for efficient resource usage and to
227 // maximize GPU utilization.
228
229 // Signal and increment the fence value.
230 const UINT64 fence = m_fenceValue;
231 ThrowIfFailed(m_commandQueue->Signal(m_fence.Get(), fence));
232 m_fenceValue++;
233
234 // Wait until the previous frame is finished.
235 if (m_fence->GetCompletedValue() < fence)
236 {
237 ThrowIfFailed(m_fence->SetEventOnCompletion(fence, m_fenceEvent));
238 WaitForSingleObject(m_fenceEvent, INFINITE);
239 }
240
241 m_frameIndex = m_swapChain->GetCurrentBackBufferIndex();
242}
