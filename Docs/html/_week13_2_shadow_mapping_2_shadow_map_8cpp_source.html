92void ShadowMap::BuildDescriptors()
93{
94 // Create SRV to resource so we can sample the shadow map in a shader program.
95 D3D12_SHADER_RESOURCE_VIEW_DESC srvDesc = {};
96 srvDesc.Shader4ComponentMapping = D3D12_DEFAULT_SHADER_4_COMPONENT_MAPPING;
97 srvDesc.Format = DXGI_FORMAT_R24_UNORM_X8_TYPELESS;
98 srvDesc.ViewDimension = D3D12_SRV_DIMENSION_TEXTURE2D;
99 srvDesc.Texture2D.MostDetailedMip = 0;
100 srvDesc.Texture2D.MipLevels = 1;
101 srvDesc.Texture2D.ResourceMinLODClamp = 0.0f;
102 srvDesc.Texture2D.PlaneSlice = 0;
103 md3dDevice->CreateShaderResourceView(mShadowMap.Get(), &srvDesc, mhCpuSrv);
104
105 // Create DSV to resource so we can render to the shadow map.
106 D3D12_DEPTH_STENCIL_VIEW_DESC dsvDesc;
107 dsvDesc.Flags = D3D12_DSV_FLAG_NONE;
108 dsvDesc.ViewDimension = D3D12_DSV_DIMENSION_TEXTURE2D;
109 dsvDesc.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
110 dsvDesc.Texture2D.MipSlice = 0;
111 md3dDevice->CreateDepthStencilView(mShadowMap.Get(), &dsvDesc, mhCpuDsv);
112}

114void ShadowMap::BuildResource()
115{
116 D3D12_RESOURCE_DESC texDesc;
117 ZeroMemory(&texDesc, sizeof(D3D12_RESOURCE_DESC));
118 texDesc.Dimension = D3D12_RESOURCE_DIMENSION_TEXTURE2D;
119 texDesc.Alignment = 0;
120 texDesc.Width = mWidth;
121 texDesc.Height = mHeight;
122 texDesc.DepthOrArraySize = 1;
123 texDesc.MipLevels = 1;
124 texDesc.Format = mFormat;
125 texDesc.SampleDesc.Count = 1;
126 texDesc.SampleDesc.Quality = 0;
127 texDesc.Layout = D3D12_TEXTURE_LAYOUT_UNKNOWN;
128 texDesc.Flags = D3D12_RESOURCE_FLAG_ALLOW_DEPTH_STENCIL;
129
130 D3D12_CLEAR_VALUE optClear;
131 optClear.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
132 optClear.DepthStencil.Depth = 1.0f;
133 optClear.DepthStencil.Stencil = 0;
134
135 ThrowIfFailed(md3dDevice->CreateCommittedResource(
136 &CD3DX12_HEAP_PROPERTIES(D3D12_HEAP_TYPE_DEFAULT),
137 D3D12_HEAP_FLAG_NONE,
138 &texDesc,
139 D3D12_RESOURCE_STATE_GENERIC_READ,
140 &optClear,
141 IID_PPV_ARGS(&mShadowMap)));
142}
