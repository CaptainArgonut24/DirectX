171 ThrowIfFailed(mCommandList->Reset(mDirectCmdListAlloc.Get(), nullptr));

174 // so we have to query this information.

179 LoadTextures();
180 BuildRootSignature();
181 BuildDescriptorHeaps();
182 BuildShadersAndInputLayout();
183 BuildShapeGeometry();
184 BuildMaterials();
185 BuildRenderItems();
186 BuildFrameResources();
187 BuildPSOs();
188
189 // Execute the initialization commands.
190 ThrowIfFailed(mCommandList->Close());
191 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
192 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
193
194 // Wait until initialization is complete.
195 FlushCommandQueue();

206void CameraApp::Update(const GameTimer& gt)
207{
208 OnKeyboardInput(gt);
209
210 // Cycle through the circular frame resource array.
211 mCurrFrameResourceIndex = (mCurrFrameResourceIndex + 1) % gNumFrameResources;
212 mCurrFrameResource = mFrameResources[mCurrFrameResourceIndex].get();
213
214 // Has the GPU finished processing the commands of the current frame resource?
215 // If not, wait until the GPU has completed commands up to this fence point.
216 if(mCurrFrameResource->Fence != 0 && mFence->GetCompletedValue() < mCurrFrameResource->Fence)
217 {
218 HANDLE eventHandle = CreateEventEx(nullptr, nullptr, false, EVENT_ALL_ACCESS);
219 ThrowIfFailed(mFence->SetEventOnCompletion(mCurrFrameResource->Fence, eventHandle));
220 WaitForSingleObject(eventHandle, INFINITE);
221 CloseHandle(eventHandle);
222 }
223
224 AnimateMaterials(gt);
225 UpdateObjectCBs(gt);
226 UpdateMaterialCBs(gt);
227 UpdateMainPassCB(gt);
228}

230void CameraApp::Draw(const GameTimer& gt)
231{
232 auto cmdListAlloc = mCurrFrameResource->CmdListAlloc;
233
234 // Reuse the memory associated with command recording.
235 // We can only reset when the associated command lists have finished execution on the GPU.
236 ThrowIfFailed(cmdListAlloc->Reset());
237
238 // A command list can be reset after it has been added to the command queue via ExecuteCommandList.
239 // Reusing the command list reuses memory.
240 ThrowIfFailed(mCommandList->Reset(cmdListAlloc.Get(), mPSOs["opaque"].Get()));
241
242 mCommandList->RSSetViewports(1, &mScreenViewport);
243 mCommandList->RSSetScissorRects(1, &mScissorRect);
244
245 // Indicate a state transition on the resource usage.
246 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
247 D3D12_RESOURCE_STATE_PRESENT, D3D12_RESOURCE_STATE_RENDER_TARGET));
248
249 // Clear the back buffer and depth buffer.
250 mCommandList->ClearRenderTargetView(CurrentBackBufferView(), Colors::LightSteelBlue, 0, nullptr);
251 mCommandList->ClearDepthStencilView(DepthStencilView(), D3D12_CLEAR_FLAG_DEPTH | D3D12_CLEAR_FLAG_STENCIL, 1.0f, 0, 0, nullptr);
252
253 // Specify the buffers we are going to render to.
254 mCommandList->OMSetRenderTargets(1, &CurrentBackBufferView(), true, &DepthStencilView());
255
256 ID3D12DescriptorHeap* descriptorHeaps[] = { mSrvDescriptorHeap.Get() };
257 mCommandList->SetDescriptorHeaps(_countof(descriptorHeaps), descriptorHeaps);
258
259 mCommandList->SetGraphicsRootSignature(mRootSignature.Get());
260
261 auto passCB = mCurrFrameResource->PassCB->Resource();
262 mCommandList->SetGraphicsRootConstantBufferView(2, passCB->GetGPUVirtualAddress());
263
264 DrawRenderItems(mCommandList.Get(), mOpaqueRitems);
265
266 // Indicate a state transition on the resource usage.
267 mCommandList->ResourceBarrier(1, &CD3DX12_RESOURCE_BARRIER::Transition(CurrentBackBuffer(),
268 D3D12_RESOURCE_STATE_RENDER_TARGET, D3D12_RESOURCE_STATE_PRESENT));
269
270 // Done recording commands.
271 ThrowIfFailed(mCommandList->Close());
272
273 // Add the command list to the queue for execution.
274 ID3D12CommandList* cmdsLists[] = { mCommandList.Get() };
275 mCommandQueue->ExecuteCommandLists(_countof(cmdsLists), cmdsLists);
276
277 // Swap the back and front buffers
278 ThrowIfFailed(mSwapChain->Present(0, 0));
279 mCurrBackBuffer = (mCurrBackBuffer + 1) % SwapChainBufferCount;
280
281 // Advance the fence value to mark commands up to this fence point.
282 mCurrFrameResource->Fence = ++mCurrentFence;
283
284 // Add an instruction to the command queue to set a new fence point.
285 // Because we are on the GPU timeline, the new fence point won't be
286 // set until the GPU finishes processing all the commands prior to this Signal().
287 mCommandQueue->Signal(mFence.Get(), mCurrentFence);
288}

359void CameraApp::UpdateObjectCBs(const GameTimer& gt)
360{
361 auto currObjectCB = mCurrFrameResource->ObjectCB.get();
362 for(auto& e : mAllRitems)
363 {
364 // Only update the cbuffer data if the constants have changed.
365 // This needs to be tracked per frame resource.
366 if(e->NumFramesDirty > 0)
367 {
368 XMMATRIX world = XMLoadFloat4x4(&e->World);
369 XMMATRIX texTransform = XMLoadFloat4x4(&e->TexTransform);
370
371 ObjectConstants objConstants;
372 XMStoreFloat4x4(&objConstants.World, XMMatrixTranspose(world));
373 XMStoreFloat4x4(&objConstants.TexTransform, XMMatrixTranspose(texTransform));
374
375 currObjectCB->CopyData(e->ObjCBIndex, objConstants);
376
377 // Next FrameResource need to be updated too.
378 e->NumFramesDirty--;
379 }
380 }
381}

383void CameraApp::UpdateMaterialCBs(const GameTimer& gt)
384{
385 auto currMaterialCB = mCurrFrameResource->MaterialCB.get();
386 for (auto& e : mMaterials)
387 {
388 // Only update the cbuffer data if the constants have changed. If the cbuffer
389 // data changes, it needs to be updated for each FrameResource.
390 Material* mat = e.second.get();
391 if (mat->NumFramesDirty > 0)
392 {
393 XMMATRIX matTransform = XMLoadFloat4x4(&mat->MatTransform);
394
395 MaterialConstants matConstants;
396 matConstants.DiffuseAlbedo = mat->DiffuseAlbedo;
397 matConstants.FresnelR0 = mat->FresnelR0;
398 matConstants.Roughness = mat->Roughness;
399 XMStoreFloat4x4(&matConstants.MatTransform, XMMatrixTranspose(matTransform));
400
401 currMaterialCB->CopyData(mat->MatCBIndex, matConstants);
402
403 // Next FrameResource need to be updated too.
404 mat->NumFramesDirty--;
405 }

409void CameraApp::UpdateMainPassCB(const GameTimer& gt)
410{
411 XMMATRIX view = mCamera.GetView();
412 XMMATRIX proj = mCamera.GetProj();
413
414 XMMATRIX viewProj = XMMatrixMultiply(view, proj);
415 XMMATRIX invView = XMMatrixInverse(&XMMatrixDeterminant(view), view);
416 XMMATRIX invProj = XMMatrixInverse(&XMMatrixDeterminant(proj), proj);
417 XMMATRIX invViewProj = XMMatrixInverse(&XMMatrixDeterminant(viewProj), viewProj);
418
419 XMStoreFloat4x4(&mMainPassCB.View, XMMatrixTranspose(view));
420 XMStoreFloat4x4(&mMainPassCB.InvView, XMMatrixTranspose(invView));
421 XMStoreFloat4x4(&mMainPassCB.Proj, XMMatrixTranspose(proj));
422 XMStoreFloat4x4(&mMainPassCB.InvProj, XMMatrixTranspose(invProj));
423 XMStoreFloat4x4(&mMainPassCB.ViewProj, XMMatrixTranspose(viewProj));
424 XMStoreFloat4x4(&mMainPassCB.InvViewProj, XMMatrixTranspose(invViewProj));
425 mMainPassCB.EyePosW = mCamera.GetPosition3f();
426 mMainPassCB.RenderTargetSize = XMFLOAT2((float)mClientWidth, (float)mClientHeight);
427 mMainPassCB.InvRenderTargetSize = XMFLOAT2(1.0f / mClientWidth, 1.0f / mClientHeight);
428 mMainPassCB.NearZ = 1.0f;
429 mMainPassCB.FarZ = 1000.0f;
430 mMainPassCB.TotalTime = gt.TotalTime();
431 mMainPassCB.DeltaTime = gt.DeltaTime();
432 mMainPassCB.AmbientLight = { 0.25f, 0.25f, 0.35f, 1.0f };
433 mMainPassCB.Lights[0].Direction = { 0.57735f, -0.57735f, 0.57735f };
434 mMainPassCB.Lights[0].Strength = { 0.8f, 0.8f, 0.8f };

480void CameraApp::BuildRootSignature()
481{
482 CD3DX12_DESCRIPTOR_RANGE texTable;
483 texTable.Init(
484 D3D12_DESCRIPTOR_RANGE_TYPE_SRV,
485 1, // number of descriptors
486 0); // register t0
487
488 // Root parameter can be a table, root descriptor or root constants.
489 CD3DX12_ROOT_PARAMETER slotRootParameter[4];
490
491 // Perfomance TIP: Order from most frequent to least frequent.
492 slotRootParameter[0].InitAsDescriptorTable(1, &texTable, D3D12_SHADER_VISIBILITY_PIXEL);
493 slotRootParameter[1].InitAsConstantBufferView(0); // register b0
494 slotRootParameter[2].InitAsConstantBufferView(1); // register b1
495 slotRootParameter[3].InitAsConstantBufferView(2); // register b2
496
497 auto staticSamplers = GetStaticSamplers();
498
499 // A root signature is an array of root parameters.
500 CD3DX12_ROOT_SIGNATURE_DESC rootSigDesc(4, slotRootParameter,
501 (UINT)staticSamplers.size(), staticSamplers.data(),
502 D3D12_ROOT_SIGNATURE_FLAG_ALLOW_INPUT_ASSEMBLER_INPUT_LAYOUT);
503
504 // create a root signature with a single slot which points to a descriptor range consisting of a single constant buffer
505 ComPtr<ID3DBlob> serializedRootSig = nullptr;
506 ComPtr<ID3DBlob> errorBlob = nullptr;
507 HRESULT hr = D3D12SerializeRootSignature(&rootSigDesc, D3D_ROOT_SIGNATURE_VERSION_1,
508 serializedRootSig.GetAddressOf(), errorBlob.GetAddressOf());
509
510 if(errorBlob != nullptr)
511 {
512 ::OutputDebugStringA((char*)errorBlob->GetBufferPointer());
513 }
514 ThrowIfFailed(hr);
515
516 ThrowIfFailed(md3dDevice->CreateRootSignature(
517 0,
518 serializedRootSig->GetBufferPointer(),
519 serializedRootSig->GetBufferSize(),
520 IID_PPV_ARGS(mRootSignature.GetAddressOf())));
521}

594void CameraApp::BuildShapeGeometry()
595{
596 GeometryGenerator geoGen;
597 GeometryGenerator::MeshData box = geoGen.CreateBox(1.0f, 1.0f, 1.0f, 3);
598 GeometryGenerator::MeshData grid = geoGen.CreateGrid(20.0f, 30.0f, 60, 40);
599 GeometryGenerator::MeshData sphere = geoGen.CreateSphere(0.5f, 20, 20);
600 GeometryGenerator::MeshData cylinder = geoGen.CreateCylinder(0.5f, 0.3f, 3.0f, 20, 20);
601
602 //
603 // We are concatenating all the geometry into one big vertex/index buffer. So
604 // define the regions in the buffer each submesh covers.
605 //
606
607 // Cache the vertex offsets to each object in the concatenated vertex buffer.
608 UINT boxVertexOffset = 0;
609 UINT gridVertexOffset = (UINT)box.Vertices.size();
610 UINT sphereVertexOffset = gridVertexOffset + (UINT)grid.Vertices.size();
611 UINT cylinderVertexOffset = sphereVertexOffset + (UINT)sphere.Vertices.size();
612
613 // Cache the starting index for each object in the concatenated index buffer.
614 UINT boxIndexOffset = 0;
615 UINT gridIndexOffset = (UINT)box.Indices32.size();
616 UINT sphereIndexOffset = gridIndexOffset + (UINT)grid.Indices32.size();
617 UINT cylinderIndexOffset = sphereIndexOffset + (UINT)sphere.Indices32.size();
618
619 SubmeshGeometry boxSubmesh;
620 boxSubmesh.IndexCount = (UINT)box.Indices32.size();
621 boxSubmesh.StartIndexLocation = boxIndexOffset;
622 boxSubmesh.BaseVertexLocation = boxVertexOffset;
623
624 SubmeshGeometry gridSubmesh;
625 gridSubmesh.IndexCount = (UINT)grid.Indices32.size();
626 gridSubmesh.StartIndexLocation = gridIndexOffset;
627 gridSubmesh.BaseVertexLocation = gridVertexOffset;
628
629 SubmeshGeometry sphereSubmesh;
630 sphereSubmesh.IndexCount = (UINT)sphere.Indices32.size();
631 sphereSubmesh.StartIndexLocation = sphereIndexOffset;
632 sphereSubmesh.BaseVertexLocation = sphereVertexOffset;
633
634 SubmeshGeometry cylinderSubmesh;
635 cylinderSubmesh.IndexCount = (UINT)cylinder.Indices32.size();
636 cylinderSubmesh.StartIndexLocation = cylinderIndexOffset;
637 cylinderSubmesh.BaseVertexLocation = cylinderVertexOffset;
638
639 //
640 // Extract the vertex elements we are interested in and pack the
641 // vertices of all the meshes into one vertex buffer.
642 //
643
644 auto totalVertexCount =
645 box.Vertices.size() +
646 grid.Vertices.size() +
647 sphere.Vertices.size() +
648 cylinder.Vertices.size();
649
650 std::vector<Vertex> vertices(totalVertexCount);
651
652 UINT k = 0;
653 for(size_t i = 0; i < box.Vertices.size(); ++i, ++k)
654 {
655 vertices[k].Pos = box.Vertices[i].Position;
656 vertices[k].Normal = box.Vertices[i].Normal;
657 vertices[k].TexC = box.Vertices[i].TexC;
658 }
659
660 for(size_t i = 0; i < grid.Vertices.size(); ++i, ++k)
661 {
662 vertices[k].Pos = grid.Vertices[i].Position;
663 vertices[k].Normal = grid.Vertices[i].Normal;
664 vertices[k].TexC = grid.Vertices[i].TexC;
665 }
666
667 for(size_t i = 0; i < sphere.Vertices.size(); ++i, ++k)
668 {
669 vertices[k].Pos = sphere.Vertices[i].Position;
670 vertices[k].Normal = sphere.Vertices[i].Normal;
671 vertices[k].TexC = sphere.Vertices[i].TexC;
672 }
673
674 for(size_t i = 0; i < cylinder.Vertices.size(); ++i, ++k)
675 {
676 vertices[k].Pos = cylinder.Vertices[i].Position;
677 vertices[k].Normal = cylinder.Vertices[i].Normal;
678 vertices[k].TexC = cylinder.Vertices[i].TexC;
679 }

682 indices.insert(indices.end(), std::begin(box.GetIndices16()), std::end(box.GetIndices16()));
683 indices.insert(indices.end(), std::begin(grid.GetIndices16()), std::end(grid.GetIndices16()));

694 CopyMemory(geo->VertexBufferCPU->GetBufferPointer(), vertices.data(), vbByteSize);
695
696 ThrowIfFailed(D3DCreateBlob(ibByteSize, &geo->IndexBufferCPU));
697 CopyMemory(geo->IndexBufferCPU->GetBufferPointer(), indices.data(), ibByteSize);
698
699 geo->VertexBufferGPU = d3dUtil::CreateDefaultBuffer(md3dDevice.Get(),
700 mCommandList.Get(), vertices.data(), vbByteSize, geo->VertexBufferUploader);
701

703 mCommandList.Get(), indices.data(), ibByteSize, geo->IndexBufferUploader);

705 geo->VertexByteStride = sizeof(Vertex);
706 geo->VertexBufferByteSize = vbByteSize;
707 geo->IndexFormat = DXGI_FORMAT_R16_UINT;
708 geo->IndexBufferByteSize = ibByteSize;
709
710 geo->DrawArgs["box"] = boxSubmesh;
711 geo->DrawArgs["grid"] = gridSubmesh;
712 geo->DrawArgs["sphere"] = sphereSubmesh;
713 geo->DrawArgs["cylinder"] = cylinderSubmesh;

715 mGeometries[geo->Name] = std::move(geo);
