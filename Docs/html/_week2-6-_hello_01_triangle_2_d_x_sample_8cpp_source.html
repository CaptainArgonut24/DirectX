41// Helper function for acquiring the first available hardware adapter that supports Direct3D 12.
42// If no such adapter can be found, *ppAdapter will be set to nullptr.
43_Use_decl_annotations_
44void DXSample::GetHardwareAdapter(IDXGIFactory2* pFactory, IDXGIAdapter1** ppAdapter)
45{
46 //Microsoft::WRL:ComPtr<> declares a type that represents the underlying interface pointer. Use ComPtr<> to declare a variable and then use the arrow member-access operator (->) to access an interface member function.
47 ComPtr<IDXGIAdapter1> adapter;
48 *ppAdapter = nullptr;
49
50 for (UINT adapterIndex = 0; DXGI_ERROR_NOT_FOUND != pFactory->EnumAdapters1(adapterIndex, &adapter); ++adapterIndex)
51 {
52 DXGI_ADAPTER_DESC1 desc;
53 adapter->GetDesc1(&desc);
54
55 if (desc.Flags & DXGI_ADAPTER_FLAG_SOFTWARE)
56 {
57 // Don't select the Basic Render Driver adapter.
58 // If you want a software adapter, pass in "/warp" on the command line.
59 continue;
60 }
61
62 // Check to see if the adapter supports Direct3D 12, but don't create the
63 // actual device yet.
64 if (SUCCEEDED(D3D12CreateDevice(adapter.Get(), D3D_FEATURE_LEVEL_11_0, _uuidof(ID3D12Device), nullptr)))
65 {
66 break;
67 }
68 }
69
70 *ppAdapter = adapter.Detach();
71}

80// Helper function for parsing any supplied command line args.
81_Use_decl_annotations_
82void DXSample::ParseCommandLineArgs(WCHAR* argv[], int argc)
83{
84 for (int i = 1; i < argc; ++i)
85 {
86 if (_wcsnicmp(argv[i], L"-warp", wcslen(argv[i])) == 0 ||
87 _wcsnicmp(argv[i], L"/warp", wcslen(argv[i])) == 0)
88 {
89 m_useWarpDevice = true;
90 m_title = m_title + L" (WARP)";
91 }
92 }
93}
