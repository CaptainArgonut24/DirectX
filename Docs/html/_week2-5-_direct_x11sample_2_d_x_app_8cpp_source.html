43int DXApp::Run()
44{
45 //Main Message Loop
46 MSG msg = { 0 };
47 while (WM_QUIT != msg.message)
48 {
49 if (PeekMessage(&msg, NULL, NULL, NULL, PM_REMOVE)) //peek and remove from the stack
50 {
51 TranslateMessage(&msg);
52 DispatchMessage(&msg);
53 }
54 else
55 {
56 //In off-time we update and render
57 Update(0.0);
58 Render(0.0);
59 }
60 }
61
62 return static_cast<int>(msg.wParam);

117bool DXApp::InitDirect3D()
118{
119 UINT createDeviceFlags = 0;
120#ifdef _DEBUG
121createDeviceFlags |= D3D11_CREATE_DEVICE_DEBUG;
122#endif //_DEBUG
123
124D3D_DRIVER_TYPE driverTypes[] =
125{
126 D3D_DRIVER_TYPE_HARDWARE, //graphics card
127 D3D_DRIVER_TYPE_WARP, //if you don't have graphics card
128 D3D_DRIVER_TYPE_REFERENCE
129};
130
131UINT numDriverTypes = ARRAYSIZE(driverTypes);
132
133
134D3D_FEATURE_LEVEL featureLevels[] =
135{
136 D3D_FEATURE_LEVEL_11_0,
137 D3D_FEATURE_LEVEL_10_1,
138 D3D_FEATURE_LEVEL_10_0,
139 D3D_FEATURE_LEVEL_9_3
140};
141UINT numFeatureLevels = ARRAYSIZE(featureLevels);
142
143DXGI_SWAP_CHAIN_DESC swapDesc;
144ZeroMemory(&swapDesc, sizeof(DXGI_SWAP_CHAIN_DESC));
145swapDesc.BufferCount = 1; //double buffer
146swapDesc.BufferDesc.Width = m_ClientWidth;
147swapDesc.BufferDesc.Height = m_ClientHeight;
148swapDesc.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
149swapDesc.BufferDesc.RefreshRate.Numerator = 60;
150swapDesc.BufferDesc.RefreshRate.Denominator = 1;
151swapDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
152swapDesc.OutputWindow = m_hAppWnd;
153swapDesc.SwapEffect = DXGI_SWAP_EFFECT_DISCARD;
154swapDesc.Windowed = true;
155swapDesc.SampleDesc.Count = 1;
156swapDesc.SampleDesc.Quality = 0;
157swapDesc.Flags = DXGI_SWAP_CHAIN_FLAG_ALLOW_MODE_SWITCH; //alt_enter full screen
158
159HRESULT result;
160for(int i=0; i< numDriverTypes; i++)
161{
162 result = D3D11CreateDeviceAndSwapChain(NULL, driverTypes[i], NULL, createDeviceFlags,
163 featureLevels, numFeatureLevels, D3D11_SDK_VERSION, &swapDesc, &m_pSwapChain, &m_pDevice,
164 &m_FeatureLevel, &m_pImmediateContext);
165
166 if (SUCCEEDED(result)) {
167 m_DriverType = driverTypes[i];
168 break;
169 }
170}
171
172if (FAILED(result)) {
173 OutputDebugString(L"Failed To Create Device and Swap chain");
174 return false;
175}
176
177//CREATE RENDER TARGET VIEW
178ID3D11Texture2D* m_pBackBufferTex = 0;
179m_pSwapChain->GetBuffer(NULL, __uuidof(ID3D11Texture2D), reinterpret_cast<void**>(&m_pBackBufferTex));
180m_pDevice->CreateRenderTargetView(m_pBackBufferTex, nullptr, &m_pRenderTargetView);
181
182//BIND RENDER TARGETVIEW
183m_pImmediateContext->OMSetRenderTargets(1, &m_pRenderTargetView, nullptr);
184
185//VIEWPORT CREATION
186m_Viewport.Width = static_cast<float> (m_ClientWidth);
187m_Viewport.Height = static_cast<float> (m_ClientHeight);
188
189m_Viewport.TopLeftX = 0;
190m_Viewport.TopLeftY = 0;
191m_Viewport.MinDepth = 0.0f;
192m_Viewport.MaxDepth = 1.0f;
193
194//BIND Viewport
195m_pImmediateContext->RSSetViewports(1, &m_Viewport);
196
197return true;
198}
